#This script load the schema for Taurant DGraph Database


type Buyer {
  buyer_id
  buyer_name
  age
  perform
}

buyer_id: string .
buyer_name: string .
age: int .
perform: [uid] .

type Transaction {
  transaction_id
  is_made_by
  device
  ip
  trade
}

transaction_id: string .
device: string .
ip: string .
is_made_by: uid @reverse .
trade: [uid] @reverse .

type Product {
  product_id
  product_name
  price
  has_transactions
}

product_id: string .
product_name: string .
price: int .
has_transactions: [uid] @count .
